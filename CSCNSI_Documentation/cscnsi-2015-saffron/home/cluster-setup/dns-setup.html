DNS setup

Intro

Domain Name System (DNS) is a naming system for computers that converts domain names to corresponding IP addresses. By putting a DNS server on our head node, we can eliminate some outside traffic and get faster responses.

Installation

[user@saffron ~]$ yum install bind
[user@saffron ~]$ yum install bind-libs
[user@saffron ~]$ yum install bind-utils

Configuration

Edit the bind config

[user@saffron ~]$ sudo vim /etc/named.conf

//

// named.conf

//

// Provided by Red Hat bind package to configure the ISC BIND named(8) DNS

// server as a caching only nameserver (as a localhost DNS resolver only).

//

// See /usr/share/doc/bind*/sample/ for example named configuration files.

//


options {

listen-on port 53 { 127.0.0.1; 192.168.0.1; };

listen-on-v6 port 53 { ::1; };

directory "/var/named";

dump-file "/var/named/data/cache_dump.db";

        statistics-file "/var/named/data/named_stats.txt";

        memstatistics-file "/var/named/data/named_mem_stats.txt";

forwarders {10.0.35.34;};

listen-on{localhost; 192.168.0.1;};

allow-query     { localhost; 192.168.0.0/16;};

recursion yes;


dnssec-enable yes;

dnssec-validation yes;

dnssec-lookaside auto;


/* Path to ISC DLV key */

bindkeys-file "/etc/named.iscdlv.key";


managed-keys-directory "/var/named/dynamic";

};


logging {

        channel default_debug {

                file "data/named.run";

                severity dynamic;

        };

};


zone "." IN {

type hint;

file "named.ca";

};


include "/etc/named.rfc1912.zones";

include "/etc/named.root.key";


zone "saffron.com" IN { #FORWARD Zone file and configuration

    type master;

    file "fwd.saffron.com";

    allow-update { none;  };

};


zone "0.168.192.in-addr.arpa" IN { # REVERSE Zone file and configuration

    type master;

    file "rev.saffron.com";

    allow-update { none; };

};


Head
/etc/resolv.conf
; generated by tom
search saffron.com
nameserver 192.168.0.1

Restart DNS

[user@saffron ~]$ sudo service rpcbind restart
[user@saffron ~]$ sudo service named restart

Punch another hole in iptables

We need to allow DNS through our firewall, so we opened up port 53 on tcp and udp.
    -A INPUT -p udp -m state --state NEW --dport 53 -j ACCEPT
    -A INPUT -p tcp -m state --state NEW --dport 53 -j ACCEPT

Restart iptables

[user@saffron ~]$ sudo service iptables restart

Add DNS zone files

Create the forward zone file:

[user@saffron ~]$ sudo vim /var/named/fwd.saffron.com

$TTL 86400

$ORIGIN saffron.com.

@    IN SOA    node1.saffron.com. root.saffron.com.(

        0 ;serial

        21600 ;refresh after 6 hours

        3600 ;retry after 1 hour

        604800 ;expire after 1 week

        86400) ;minimum TTL of 1 day

@     IN NS    node1.saffron.com.

node1 IN A     192.168.0.1

n01   IN A     192.168.0.1

node2 IN A     192.168.0.2

node3 IN A     192.168.0.3

node4 IN A     192.168.0.4

node5 IN A     192.168.0.5

node6 IN A     192.168.0.6

node7 IN A     192.168.0.7

node8 IN A     192.168.0.8

node9 IN A     192.168.0.9

node10 IN A     192.168.0.10

node11 IN A     192.168.0.11

ntp   IN A     192.168.0.1

nfs   IN A     192.168.0.1

dns   IN A     192.168.0.1

@     IN A     192.168.0.1

www   IN A     192.168.0.1

ldap  IN A     192.168.0.1

ibn1  IN A    192.168.2.1

ibn2  IN A    192.168.2.2

ibn4  IN A    192.168.2.4

Create the reverse lookup zone file:

[user@saffron ~]$ sudo vim /var/named/rev.saffron.com

$TTL 86400

@    IN SOA    node1.saffron.com. root.saffron.com.(

        0 ;serial

        1D ;refresh

        1H ;retry

        1W ;expire

        3H) ;minimum

@     IN NS    node1.saffron.com.

0.1     IN PTR   node1.saffron.com.

0.2     IN PTR   node2.saffron.com.

0.3     IN PTR   node3.saffron.com.

Test the config files:

[user@saffron ~]$ chkconfig named on
[user@saffron ~]$ named-checkconf /etc/named.conf
[user@saffron ~]$ named-checkzone saffron.com /var/named/fwd.saffron.com
[user@saffron ~]$ named-checkzone 0.168.192.in-addr.arpa /var/named/rev.saffron.com
